I. โค้ดตัวอย่างใน Java, C++, Python พร้อมผลรัน

1. Java

class Animal {
    Animal() {
        System.out.println("create animal");
    }

    void sound() {
        System.out.println("animal makes a sound");
    }

    protected void finalize() {
        System.out.println("delete animal");
    }
}

class Dog extends Animal {
    Dog() {
        System.out.println("dog");
    }

    @Override
    void sound() {
        System.out.println("dog barks");
    }

    protected void finalize() {
        System.out.println("delete dog");
    }
}

class Cat extends Animal {
    Cat() {
        System.out.println("cat");
    }

    @Override
    void sound() {
        System.out.println("cat meows");
    }

    protected void finalize() {
        System.out.println("delete cat");
    }
}

public class Main {
    public static void main(String[] args) {
        Animal a = new Animal();
        Dog d = new Dog();
        Cat c = new Cat();

        a.sound();
        d.sound();
        c.sound();  
    }
}

2. C++

#include <iostream>
using namespace std;

class Animal {
public:
    Animal() {
        cout << "create animal" << endl;
    }

    virtual void sound() {
        cout << "animal makes a sound" << endl;
    }

    virtual ~Animal() {
        cout << "delete animal" << endl;
    }
};

class Dog : public Animal {
public:
    Dog() {
        cout << "dog" << endl;
    }

    void sound() override {
        cout << "dog barks" << endl;
    }

    ~Dog() {
        cout << "delete dog" << endl;
    }
};

class Cat : public Animal {
public:
    Cat() {
        cout << "cat" << endl;
    }

    void sound() override {
        cout << "cat meows" << endl;
    }

    ~Cat() {
        cout << "delete cat" << endl;
    }
};

int main() {
    Animal* a = new Animal();
    Dog* d = new Dog();
    Cat* c = new Cat();

    a->sound();
    d->sound();
    c->sound();

    delete a;
    delete d;
    delete c;

    return 0;
}

3. Python

class Animal:
    # Constructor
    def __init__(self):
        print("create animal")

    # Destructor
    def __del__(self):
        print("delete animal")

    # Method sound
    def sound(self):
        print("animal makes a sound")

class Dog(Animal):
    # Constructor
    def __init__(self):
        super().__init__()  # เรียกคอนสตรัคเตอร์ของ Animal
        print("dog")

    # Destructor
    def __del__(self):
        print("delete dog")

    # Override sound
    def sound(self):
        print("dog barks")

class Cat(Animal):
    # Constructor
    def __init__(self):
        super().__init__()  # เรียกคอนสตรัคเตอร์ของ Animal
        print("cat")

    # Destructor
    def __del__(self):
        print("delete cat")

    # Override sound
    def sound(self):
        print("cat meows")

# Create objects
a = Animal()
d = Dog()
c = Cat()

a.sound()
d.sound()
c.sound()

II. เปรียบเทียบโค้ดกับ OOP Concepts

1. Class

Class คือโครงสร้างที่เป็น blueprint สำหรับการสร้าง object เช่นในโค้ดด้านบน มีการสร้าง class Animal, Dog, และ Cat เพื่อเป็นแบบแผนสำหรับ object ที่จะสร้าง

class Animal {
    void sound() {
        System.out.println("animal makes a sound");
    }
}

2. Object, Instance

Object คือ instance ของ class ซึ่งถูกสร้างขึ้นจาก class เช่นในโค้ด Animal a = new Animal(); สร้าง object a จาก class Animal

Animal a = new Animal();
a.sound();

3. Subclass, Derived Class

Subclass หรือ Derived Class คือคลาสที่สืบทอดจาก superclass หรือ base class ตัวอย่างเช่น Dog และ Cat เป็น subclass ของ Animal

class Dog extends Animal {
    @Override
    void sound() {
        System.out.println("dog barks");
    }
}

4. Message

Message คือการส่งคำสั่งไปยัง object เพื่อให้ทำงานตาม method ที่กำหนด เช่น a.sound() ส่งคำสั่งให้ object a เรียกใช้ method sound()

a.sound();

5. Inheritance

Inheritance คือการสืบทอดคุณสมบัติจาก class หนึ่งไปยังอีก class หนึ่ง ตัวอย่างเช่น Dog และ Cat สืบทอดจาก Animal ทำให้สามารถใช้คุณสมบัติของ Animal ได้

class Dog extends Animal {
    Dog() {
        System.out.println("dog");
    }
}

6. Polymorphism

Polymorphism คือการที่ object สามารถมี method ชื่อเดียวกันแต่ทำงานต่างกัน ตัวอย่างเช่น method sound() ใน Dog, Cat และ Animal ทำงานแตกต่างกันตาม subclass

Animal a = new Dog();
a.sound(); // จะพิมพ์ว่า "dog barks"

III. Abstract Class

Abstract class คือคลาสที่ไม่สามารถสร้างวัตถุได้โดยตรง แต่สามารถสืบทอดไปใช้ในคลาสลูกได้ เพื่อบังคับให้คลาสลูก implement เมธอดที่เป็น abstract ในคลาสแม่

หลักการสร้าง abstract class:

  • ใน Java ใช้คำสั่ง abstract
  • ใน C++ ใช้คำสั่ง virtual และเมธอดแบบ pure virtual function
  • ใน Python ใช้คำสั่ง @abstractmethod

ไม่สามารถสร้างวัตถุจาก abstract class ได้โดยตรง แต่สามารถสร้างคลาสลูกที่สืบทอดได้

ตัวอย่างใน Java

abstract class Animal {
    abstract void sound(); // abstract method
}

class Dog extends Animal {
    void sound() {
        System.out.println("dog barks");
    }
}

ตัวอย่างใน C++

class Animal {
public:
    virtual void sound() = 0; // pure virtual function
};

class Dog : public Animal {
public:
    void sound() override {
        cout << "dog barks" << endl;
    }
};

ตัวอย่างใน Python

from abc import ABC, abstractmethod

class Animal(ABC):
    @abstractmethod
    def sound(self):
        pass

class Dog(Animal):
    def sound(self):
        print("dog barks")
